def plotHyperplane(w1,w2,b):
    x_1 = np.linspace(-2,4,10)
    x_2 = -(w1*x_1 + b)/w2
    x_p = -(w1*x_1 + b+1)/w2
    x_n = -(w1*x_1 + b-1)/w2
    plt.figure(figsize=((5,5)))
    plt.plot(x_1,x_2, label = "Hyperplane")
    plt.plot(x_1,x_p, label = "Positive Hyperplane")
    plt.plot(x_1,x_n, label = "Negative Hyperplane")
